HOW TO ADD NEW ADMIN PAGE:
================================================================

Template for new page:

<?php
require_once __DIR__ . '/../../config.php';
if (!isAdmin()) redirect('/admin/login.php');

// Your page logic here
$data = $pdo->query("SELECT * FROM your_table")->fetchAll();

$page_title = 'Your Page Title - Admin';
include __DIR__ . '/../includes/admin-header.php';
?>

<div class="header">
    <h1>Your Page Title</h1>
    <a href="/admin/your-page/add.php" class="btn btn-primary">+ Add New</a>
</div>

<div class="content-container">
    <table>
        <thead>
            <tr>
                <th>Column 1</th>
                <th>Column 2</th>
                <th>Actions</th>
            </tr>
        </thead>
        <tbody>
            <?php foreach ($data as $item): ?>
                <tr>
                    <td><?php echo htmlspecialchars($item['name']); ?></td>
                    <td><?php echo htmlspecialchars($item['value']); ?></td>
                    <td>
                        <a href="/admin/your-page/edit.php?id=<?php echo $item['id']; ?>" class="btn btn-secondary">Edit</a>
                    </td>
                </tr>
            <?php endforeach; ?>
        </tbody>
    </table>
</div>

<?php include __DIR__ . '/../includes/admin-footer.php'; ?>
